## 六、命名空间

很多语言里都有命名空间的概念，比如OC里的命名空间就是前缀，Swift里的命名空间就是模块，Dart里的命名空间就是包，C++里的命名空间则是namespace。

命名空间一般用来解决变量名、函数名、类名等标识符的命名冲突问题。

* 定义命名空间

> namespace 命名空间名 {
>     // 代码...
> }

```c++
// 命名空间ine
namespace ine {
    int i = 10;

    void test() {
        cout << "ine::test()" << endl;
    }

    class Person {
    public:
        Person() {
            cout << "ine::Person()" << endl;
        }
        
        ~Person() {
            cout << "ine::~Person()" << endl;
        }
    };
}

// 命名空间yee
namespace yee {
    int i = 20;

    void test() {
        cout << "yee::test()" << endl;
    }

    class Person {
    public:
        Person() {
            cout << "yee::Person()" << endl;
        }
        
        ~Person() {
            cout << "yee::~Person()" << endl;
        }
    };
}
```

* 使用命名空间

  * 整体使用命名空间法

  > using namespace 命名空间名;

  ```c++

              cout << "ine::Person()" << endl;
          }
          
          ~Person() {
              cout << "ine::~Person()" << endl;
          }
      };
  }
  
  // 命名空间yee
  namespace yee {
      int i = 20;
  
      void test() {
          cout << "yee::test()" << endl;

      class Person {
      public:
          Person() {
              cout << "yee::Person()" << endl;

  using namespace ine;

  ```

  * 独立使用命名空间法

  > 命名空间名::代码

  ```c++
  #include <iostream>
  using namespace std;
  
  // 命名空间ine
  namespace ine {
      int i = 10;
  
      void test() {
          cout << "ine::test()" << endl;
      }
  
      class Person {
      public:
          Person() {
              cout << "ine::Person()" << endl;
          }
          
          ~Person() {
              cout << "ine::~Person()" << endl;
          }
      };
  }
  
  // 命名空间yee
  namespace yee {
      int i = 20;
  
      void test() {
          cout << "yee::test()" << endl;
      }
  
      class Person {
      public:
          Person() {
              cout << "yee::Person()" << endl;
          }
          
          ~Person() {
              cout << "yee::~Person()" << endl;
          }
      };
  }
  
  int main() {
      cout << ine::i << endl; // 10
      ine::test(); // ine::test()
      ine::Person *person = new ine::Person(); // ine::Person()
      delete person; // ine::~Person()
      
      cout << yee::i << endl; // 20
      yee::test(); // yee::test()
      yee::Person *person1 = new yee::Person(); // yee::Person()
      delete person1; // yee::~Person()
      
      return 0;
  }
  ```
