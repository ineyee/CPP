## 四、多线程

#### 1、基本使用

```c++
#include <iostream>
#include <thread>
using namespace std;

// 我们创建一个子线程来改变x的值
int x = 0;

// 子线程里想做的事
// 改变[count]次x
void add(int count) {
    cout << "子线程id：" << this_thread::get_id()  << endl;
    
    for (int i = 0; i < count; i++) {
        x++;
    }
}

int main() {
    cout << "main()函数开始，主线程id：" << this_thread::get_id() << endl;
    
    // 创建子线程对象并启动
    // 第一个参数可以是个函数名，也可以是个匿名函数
    // 第二个参数是函数的参数，是个可变参数，可以传多个、没有就不传
    thread *t = new thread(add, 1000);
    
    // 等待子线程t执行完
    // 这个demo如果不等待的话，主线程就直接return 0程序结束了
    t->join();
    
    delete t;
    
    cout << x << endl;
    
    cout << "main()函数结束，主线程id：" << this_thread::get_id()  << endl;
    
    // 控制台打印：
    // main()函数开始，主线程id：0x2000c02c0
    // 子线程id：0x304195000
    // 1000
    // main()函数结束，主线程id：0x2000c02c0（被t->join()卡住了，所以后打印）
    
    return 0;
}
```

#### 2、数据竞争问题

使用多线程就必然会存在数据竞争问题。

```c++
#include <iostream>
#include <thread>
using namespace std;

// 我们创建两个子线程来改变x的值
int x = 0;

// 子线程里想做的事
// 改变[count]次x
void add(int count) {
    cout << "子线程id：" << this_thread::get_id()  << endl;
    
    for (int i = 0; i < count; i++) {
        x++;
    }
}

int main() {
    cout << "main()函数开始，主线程id：" << this_thread::get_id() << endl;
    
    thread *t = new thread(add, 1000);
    thread *t1 = new thread(add, 1000);
    
    t->join();
    t1->join();
    
    delete t;
    delete t1;
    
    cout << x << endl;
    
    cout << "main()函数结束，主线程id：" << this_thread::get_id()  << endl;
    
    // 控制台打印：
    // main()函数开始，主线程id：0x2000c02c0
    // 子线程id：0x304195000
    // 子线程id：0x304218000
    // 1175（本来应该是2000，但是因为数据竞争问题导致数据出错）
    // main()函数结束，主线程id：0x2000c02c0
    
    return 0;
}
```

#### 3、线程同步方案

处理数据竞争问题我们称之为线程同步，线程同步方案之一就是加锁。

```c++
#include <iostream>
#include <thread>
#include <mutex>
using namespace std;

// 创建一个全局的互斥锁，多个线程要用同一把锁才行
mutex mtx;

// 我们创建两个子线程来改变x的值
int x = 0;

// 子线程里想做的事
// 改变[count]次x
void add(int count) {
    cout << "子线程id：" << this_thread::get_id()  << endl;
    
    // 这种加锁方式相当于是把两个线程搞成串行的了，即线程t完整地把任务————整个for循环————执行完，线程t1才能开始执行任务
    // 加锁
    mtx.lock();
    for (int i = 0; i < count; i++) {
        x++;
    }
    // 解锁
    mtx.unlock();
}

int main() {
    cout << "main()函数开始，主线程id：" << this_thread::get_id() << endl;
    
    thread *t = new thread(add, 1000);
    thread *t1 = new thread(add, 1000);
    
    t->join();
    t1->join();
    
    delete t;
    delete t1;
    
    cout << x << endl;
    
    cout << "main()函数结束，主线程id：" << this_thread::get_id()  << endl;
    
    // 控制台打印：
    // main()函数开始，主线程id：0x2000c02c0
    // 子线程id：0x304195000
    // 子线程id：0x304218000
    // 2000
    // main()函数结束，主线程id：0x2000c02c0
    
    return 0;
}
```

#### 4、并发||并行任务同步方案

并发||并行任务同步是指在使用多线程的前提下，我们通过设置多个并发||并行任务之间的依赖关系来使得多个并发||并行任务能按照我们希望的执行顺序来同步执行，并发||并行任务同步方案之一就是条件变量 + 锁。

举个例子，我们创建两个子线程来打印[0, n)之间的整数，一个子线程用来打印偶数，另一个子线程用来打印奇数，要求先打印偶数、接下来就奇数偶数一个一个交替打印。

```c++
#include <iostream>
#include <thread>
using namespace std;

int main() {
    int n = 10;
    
    // 打印偶数线程
    thread *t_even = new thread([n] () -> void {
        for (int i = 0; i < n; i += 2) {
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
        }
    });
    
    // 打印奇数线程
    thread *t_odd = new thread([n] () -> void {
        for (int i = 1; i < n; i += 2) {
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
        }
    });
    
    
    t_even->join();
    t_odd->join();
    
    delete t_even;
    delete t_odd;
    
    // 控制台打印：
    // 子线程id：0x304195000===>：子线程id：0
    // 0x304218000子线程id：0x304195000===>：1
    // 子线程id：0x304218000===>：3
    // 子线程id：0x304218000===>：5
    // 子线程id：0x304218000===>：7
    // 子线程id：0x304218000===>：9
    // ===>：2
    // 子线程id：0x304195000===>：4
    // 子线程id：0x304195000===>：6
    // 子线程id：0x304195000===>：8
    
    return 0;
}
```

像上面这样简单地创建两个子线程来搞绝对不行，两个子线程里任务的执行绝对是并发||并行的，这从乱七八糟的控制台打印也可看出，我们得加条件变量 + 锁。

```c++
#include <iostream>
#include <thread>
#include <condition_variable>
#include <mutex>
using namespace std;

int main() {
    int n = 10;
    
    // 条件变量和锁我们不需要自己初始化，编译器会自动初始化好，我们直接拿着用就行
    // 这一对专门用来负责打印偶数线程的wait和notify
    condition_variable cv_even;
    mutex mtx_even;
    // 这一对专门用来负责打印奇数线程的wait和notify
    condition_variable cv_odd;
    mutex mtx_odd;
    
    // 打印偶数线程
    // 一次打印相当于一个任务
    // 当然匿名函数内部无法直接使用外界的局部变量，所以我们得捕获一下它们，这里编译器要求条件变量和锁必须用地址捕获
    thread *t_even = new thread([n, &cv_even, &mtx_even, &cv_odd] () -> void {
        for (int i = 0; i < n; i += 2) {
            // 因为要求先打印偶数，所以这个线程一进来就应该开始执行任务————即开始打印，同时还必须得让打印奇数线程一进去就开始等待
            // 这样无论是这个线程先执行、还是这个线程和打印奇数线程同时执行、还是这个线程比打印奇数线程后执行，都可以保证先执行这里的任务————即先打印偶数
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
            
            // 打印完偶数，通知一下打印奇数线程的wait可以往下执行了
            // 唤醒哪个线程就用哪个线程对应的条件变量 + 锁
            cv_odd.notify_one();
            
            // 但是notify_one()这个函数是异步的，不会阻塞，所以这里我们需要立即开始等待，等待打印奇数线程打印一个奇数，否则这里就会立马执行下一次循环打印下一个偶数了
            // 让哪个线程等待就用哪个线程对应的条件变量 + 锁
            auto ul = unique_lock<mutex>(mtx_even);
            cv_even.wait(ul);
        }
    });
    
    // 打印奇数线程
    // 一次打印相当于一个任务
    // 当然匿名函数内部无法直接使用外界的局部变量，所以我们得捕获一下它们，这里编译器要求条件变量和锁必须用地址捕获
    thread *t_odd = new thread([n, &cv_odd, &mtx_odd, &cv_even] () -> void {
        for (int i = 1; i < n; i += 2) {
            // 因为要求先打印偶数，所以这个线程一进来就应该开始等待，而不应该开始执行任务————即开始打印，因为有可能是这个线程先执行或者这个线程和打印偶数线程同时执行，如果不等待的话就会先打印奇数了
            // 让哪个线程等待就用哪个线程对应的条件变量 + 锁
            auto ul = unique_lock<mutex>(mtx_odd);
            cv_odd.wait(ul);
            
            // 执行任务————即打印奇数
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
            
            // 打印完奇数，通知一下打印偶数线程的wait可以往下执行了
            // 唤醒哪个线程就用哪个线程对应的条件变量 + 锁
            cv_even.notify_one();
        }
    });
    
    
    t_even->join();
    t_odd->join();
    
    delete t_even;
    delete t_odd;
    
    // 控制台打印：
    // 子线程id：0x304195000===>：0
    // 子线程id：0x304218000===>：1
    // 子线程id：0x304195000===>：2
    // 子线程id：0x304218000===>：3
    // 子线程id：0x304195000===>：4
    // 子线程id：0x304218000===>：5
    // 子线程id：0x304195000===>：6
    // 子线程id：0x304218000===>：7
    // 子线程id：0x304195000===>：8
    // 子线程id：0x304218000===>：9
    
    return 0;
}
```

#### 5、死锁

所谓死锁就是指两个或多个线程在互相等待对方线程解锁，从而导致大家都解不了锁，进而导致程序卡死在那儿了，开发中我们要避免死锁的发生。

还是上面的例子，我们稍微修改下代码就会造成死锁，两个线程在互相等待对方发通知，导致都无法往下执行。

```c++
#include <iostream>
#include <thread>
#include <condition_variable>
#include <mutex>
using namespace std;

int main() {
    int n = 10;
    
    condition_variable cv_even;
    mutex mtx_even;
    condition_variable cv_odd;
    mutex mtx_odd;
    
    thread *t_even = new thread([n, &cv_even, &mtx_even, &cv_odd] () -> void {
        for (int i = 0; i < n; i += 2) {
            // 等打印奇数线程发通知...
            auto ul = unique_lock<mutex>(mtx_even);
            cv_even.wait(ul);
            
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
            
            cv_odd.notify_one();
        }
    });
    
    thread *t_odd = new thread([n, &cv_odd, &mtx_odd, &cv_even] () -> void {
        for (int i = 1; i < n; i += 2) {
            // 等打印偶数线程发通知...
            auto ul = unique_lock<mutex>(mtx_odd);
            cv_odd.wait(ul);
            
            cout << "子线程id：" << this_thread::get_id() << "===>：" << i << endl;
            
            cv_even.notify_one();
        }
    });
    
    
    t_even->join();
    t_odd->join();
    
    delete t_even;
    delete t_odd;
    
    return 0;
}
```
